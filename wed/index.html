using namespace std;

const int red = 1;  //GPIO 1
const int yellow = 4;   //GPIO 4
const int green = 5;  //GPIO 5

int main(){
    if (wiringPiSetup() == -1){ 
        return 0;
    }

    pinMode(red, OUTPUT);
    pinMode(yellow, OUTPUT);
    pinMode(green, OUTPUT);

    while (1){
        //green 3s
        digitalWrite(red, 0);
        digitalWrite(yellow, 0);
        digitalWrite(green, 1);
        delay(3000);

        //green flash x3
        digitalWrite(red, 0);
        digitalWrite(yellow, 0);
        
        for (int i = 0; i < 3; i++){
            digitalWrite(green, 1);
            delay(500);
            digitalWrite(green, 0);
            delay(500);
        }

        //yellow 2s
        digitalWrite(red, 0);
        digitalWrite(yellow, 1);
        digitalWrite(green, 0);
        delay(2000);

        //red 3s
        digitalWrite(red, 1);
        digitalWrite(yellow, 0);
        digitalWrite(green, 0);
        delay(3000);
        
        digitalWrite(red, 0);

    }
}
